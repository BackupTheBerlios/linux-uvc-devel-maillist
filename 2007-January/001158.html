OpenCV prog:

int main(int argc, char* argv[])
{
        IplImage * frame = NULL;
        CvCapture *input_cam = cvCaptureFromCAM(-1);

        frame = cvQueryFrame(input_cam);

        cvNamedWindow("capture", 0);
        cvResizeWindow("capture", frame->width, frame->height);

        while(frame != NULL){
                cvShowImage("capture", frame);
                cvWaitKey(10);
                frame = cvQueryFrame(input_cam);
        }
        return 0;
}
